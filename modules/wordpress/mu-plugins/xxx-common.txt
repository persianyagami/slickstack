add_action('admin_bar_menu', 'ss_environment_menu', 25);

function ss_environment_menu($admin_bar) {
    
    // global $wp_admin_bar;
    
    $current_user = wp_get_current_user();
    
    $ss_production_link = home_url( $path = '', $scheme = 'https' );
    // $ss_production_tld =  preg_replace( '|https?://[^/]+|i', '', $ss_production_link ) );
    
    $ss_remove_staging_directory = '/staging';	
    $ss_remove_dev_directory = '/dev';	
    
    $ss_staging_prefix = 'https://staging.';
    $ss_staging_directory = '/staging/';
    $ss_dev_prefix = 'https://dev.';
    $ss_dev_directory = '/dev/';
    
    // $ss_staging_link = $ss_production_link . $ss_staging_directory;
    
    $ss_staging_link = str_replace('https://', 'https://staging.', $ss_production_link);
    $ss_staging_link_fixed = str_replace('www.', '', $ss_staging_link);
    
    // $ss_dev_link = $ss_production_link . $ss_dev_directory;
    
    $ss_dev_link = str_replace('https://', 'https://dev.', $ss_production_link);
    $ss_dev_link_fixed = str_replace('www.', '', $ss_dev_link);
    
    $ss_production_link_trimmed_staging = str_replace($ss_remove_staging_directory, '', $ss_production_link);
    $ss_dev_link_trimmed_staging = $ss_production_link_trimmed_staging . $ss_dev_directory;
    
    $ss_production_link_trimmed_dev = str_replace($ss_remove_dev_directory, '', $ss_production_link);
    $ss_staging_link_trimmed_dev = $ss_production_link_trimmed_dev . $ss_staging_directory;
    
    if(current_user_can('manage_options')) {
    
        if(WP_ENVIRONMENT_TYPE == 'staging') {
        
            // current environment
            $admin_bar->add_menu( array(
                'parent' => 'top-secondary',
                'id'    => 'ss_environment_current',
                'title' => '<span style="color:#FFBA00;">STAGING</span>',
                'href'  => false
                ) );
            
        // Production Link
        $admin_bar->add_menu( array(
            'parent' => 'ss_environment_current', 
            'id' => 'ss_environment_toggle_production',
            'title' => 'Switch To Production', 
            'href' => $ss_production_link_trimmed_staging, 
            'meta' => array('target' => '_blank')
            ) );
            
        // Dev Link
        $admin_bar->add_menu( array(
            'parent' => 'ss_environment_current', 
            'id' => 'ss_environment_toggle_dev',
            'title' => 'Switch To Dev', 
            'href' => $ss_dev_link_trimmed_staging, 
            'meta' => array('target' => '_blank')
            ) );
            
        } // end if staging
        
        elseif(WP_ENVIRONMENT_TYPE == 'development') {
                
        // Current Environment
        $admin_bar->add_menu( array(
            'parent' => 'top-secondary',
            'id'    => 'ss_environment_current',
            'title' => '<span style="color:#dc3232;">DEVELOPMENT</span>',
            'href'  => false
            ) );
            
        // Production Link
        $admin_bar->add_menu( array(
            'parent' => 'ss_environment_current', 
            'id' => 'ss_environment_toggle_production',
            'title' => 'Switch To Production', 
            'href' => $ss_production_link_trimmed_dev, 
            'meta' => array('target' => '_blank')
            ) );
            
        // Dev Link
        $admin_bar->add_menu( array(
            'parent' => 'ss_environment_current', 
            'id' => 'ss_environment_toggle_dev',
            'title' => 'Switch To Staging', 
            'href' => $ss_staging_link_trimmed_dev, 
            'meta' => array('target' => '_blank')
            ) );
            
        } // end elseif dev
        
        else {
        
        // Current Environment
        $admin_bar->add_menu( array(
            'parent' => 'top-secondary',
            'id'    => 'ss_environment_current',
            'title' => '<span style="color:#3CB54C;">PRODUCTION</span>',
            'href'  => false
            ) );
            
        // Staging Link
        $admin_bar->add_menu( array(
            'parent' => 'ss_environment_current', 
            'id' => 'ss_environment_toggle_staging',
            'title' => 'Switch To Staging', 
            'href' => $ss_staging_link_fixed, 
            'meta' => array('target' => '_blank')
            ) );
            
        // Dev Link
        $admin_bar->add_menu( array(
            'parent' => 'ss_environment_current', 
            'id' => 'ss_environment_toggle_dev',
            'title' => 'Switch To Dev', 
            'href' => $ss_dev_link_fixed, 
            'meta' => array('target' => '_blank')
            ) );
            
        } // end else (production)
        
    } else { }
}
